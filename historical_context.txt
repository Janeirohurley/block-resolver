;
    else if (filled === 5) score += 2;
  }
  return score;
}

/**
 * Évalue si les blocs restants de la main peuvent encore être placés sur la grille.
 * Pénalise les placements qui bloquent les blocs futurs.
 */
function placementFeasibility(
  grid: boolean[][],
  remainingBlocks: BlockInstance[]
): number {
  let canPlaceCount = 0;
  for (const block of remainingBlocks) {
    const transforms = getUniqueTransforms(block.definition, block.color);
    let blockCanBePlaced = false;
    outer: for (const t of transforms) {
      const shape = getInstanceShape(t);
      for (let r = 0; r < GRID_SIZE; r++) {
        for (let c = 0; c < GRID_SIZE; c++) {
          if (canPlace(grid, shape, r, c)) {
            blockCanBePlaced = true;
            break outer;
          }
        }
      }
    }
    if (blockCanBePlaced) canPlaceCount++;
  }
  // Récompense pour chaque bloc restant qui peut encore être placé
  return canPlaceCount * 15;
}

/**
 * Heuristique composite d'évaluation d'un état de grille.
 * Score élevé = configuration favorable.
 */
function evaluateGridState(
  grid: boolean[][],
  remainingBlocks: BlockInstance[] = []
): number {
  const holes = countHoles(grid);
  const regions = countVoidRegions(grid);
  const nearComplete = nearCompleteScore(grid);
  const feasibility = placementFeasibility(grid, remainingBlocks);

  // Occupation totale (densité) — équilibrer remplissage et espace libre
  const totalFilled = grid.flat().filter(Boolean).length;
  const density = totalFilled / 64;
  // Bonus modéré pour densité basse (beaucoup d'espace libre = plus de flexibilité)
  const densityBonus = density < 0.5 ? (0.5 - density) * 10 : 0;

  return (
    nearComplete           // bonus lignes/cols proches de la complétion
    - holes * 4            // pénalité trous
    - (regions - 1) * 6   // pénalité fragmentation (1 région = parfait)
    + feasibility          // bonus si blocs restants peuvent être placés
    + densityBonus
  );
}

function applyPlacement(
  grid: boolean[][],
  shape: number[][],
  row: number,
  col: number
): boolean[][] {
  const newGrid = grid.map(r => [...r]);
  for (const [dr, dc] of shape) {
    newGrid[row + dr][col + dc] = true;
  }
  // Libérer les lignes complètes
  for (let r = 0; r < GRID_SIZE; r++) {
    if (newGrid[r].every(Boolean)) {
      for (let c = 0; c < GRID_SIZE; c++) newGrid[r][c] = false;
    }
  }
  // Libérer les colonnes complètes
  for (let c = 0; c < GRID_SIZE; c++) {
    if (newGrid.every(r => r[c])) {
      for (let r = 0; r < GRID_SIZE; r++) newGrid[r][c] = false;
    }
  }
  return newGrid;
}

// ─── Score look-ahead multi-coups ───────────────────────────────────────────

/**
 * Évalue un placement en simulant les coups suivants des blocs restants.
 * depth=1 : regarde 1 coup en avance (blocs restants dans la main).
 */
function lookAheadScore(
  gridAfterPlacement: boolean[][],
  remainingBlocks: BlockInstance[],
  depth: number
): number {
  if (depth === 0 || remainingBlocks.length === 0) {
    return evaluateGridState(gridAfterPlacement, remainingBlocks);
  }

  const [nextBlock, ...rest] = remainingBlocks;
  const transforms = getUniqueTransforms(nextBlock.definition, nextBlock.color);
  let bestScore = -Infinity;
  let hasAnyPlacement = false;

  for (const transform of transforms) {
    const shape = getInstanceShape(transform);
    for (let r = 0; r < GRID_SIZE; r++) {
      for (let c = 0; c < GRID_SIZE; c++) {
        if (canPlace(gridAfterPlacement, shape, r, c)) {
          hasAnyPlacement = true;
          const nextGrid = applyPlacement(gridAfterPlacement, shape, r, c);
          // Récursion pour regarder plus loin (depth-1)
          const score = lookAheadScore(nextGrid, rest, depth - 1);
          if (score > bestScore) bestScore = score;
        }
      }
    }
  }

  // Si aucun placement possible pour ce bloc → grosse pénalité (game over proche)
  if (!hasAnyPlacement) return -200;

  return bestScore;
}

/**
 * Génère les suggestions pour un bloc donné en tenant compte des blocs restants.
 */
function suggestionsForBlock(
  block: BlockInstance,
  grid: boolean[][],
  remainingBlocks: BlockInstance[],
  maxSuggestions = 5
): Suggestion[] {
  const suggestions: Suggestion[] = [];
  const transforms = getUniqueTransforms(block.definition, block.color);

  for (const transform of transforms) {
    const shape = getInstanceShape(transform);

    for (let row = 0; row < GRID_SIZE; row++) {
      for (let col = 0; col < GRID_SIZE; col++) {
        if (!canPlace(grid, shape, row, col)) continue;

        const { linesCleared, colsCleared, clearedLines, clearedCols } =
          calculatePlacementScore(grid, shape, row, col);

        // Grille après ce placement
        const gridAfter = applyPlacement(grid, shape, row, col);

        // Score immédiat: lignes/colonnes libérées
        const immediateScore =
          (linesCleared + colsCleared) * GRID_SIZE * 15 + shape.length * 2;

        // Score look-ahead: évalue l'avenir avec les blocs restants (depth=1)
        const futureScore = lookAheadScore(gridAfter, remainingBlocks, 1);

        // Score total pondéré (immédiat prévaut légèrement sur futur)
        const totalScore = immediateScore * 1.5 + futureScore;

        suggestions.push({
          id: `${transform.definition.id}-${transform.rotation}-${transform.flipped}-${row}-${col}`,
          blockInstance: transform,
          position: { row, col },
          score: Math.round(totalScore),
          linesCleared,
          colsCleared,
          cellsFreed: (linesCleared + colsCleared) * GRID_SIZE,
          affectedCells: shape.map(([dr, dc]) => [row + dr, col + dc] as [number, number]),
          clearedLines,
          clearedCols,
        });
      }
    }
  }

  return suggestions
    .sort((a, b) => b.score - a.score)
    .slice(0, maxSuggestions);
}

export function generateSuggestions(
  grid: Grid,
  hand: Array<BlockInstance | null>,
  maxPerBlock = 3

  const result: Record<number, Suggestion[]> = {};

          comboLabel = `🎯 Setup ×${s.nearCompleteCount}`;
        } else if (s.nearCompleteCount >= 2) {
          comboLabel = `🎯 Setup ×${s.nearCompleteCount}`;
        } else if (s.bestFutureClears >= 2) {
          comboLabel = `🔮 Futur ×${s.bestFutureClears}`;
        } else if (totalFutureLines >= 3) {
          comboLabel = `📐 ${totalFutureLines} prêts`;
        }

        candidates.push({
          id: `${transform.definition.id}-${transform.rotation}-${transform.flipped}-${row}-${col}`,
          blockInstance: transform,
          position: { row, col },
          score: Math.round(s.totalScore),
          linesCleared: s.linesCleared,
          colsCleared: s.colsCleared,
          cellsFreed: (s.linesCleared + s.colsCleared) * GRID_SIZE,
          affectedCells: shape.map(([dr, dc]) => [row + dr, col + dc] as [number, number]),
          clearedLines: s.clearedLines,
          clearedCols: s.clearedCols,
          futureComboLines: s.nearCompleteCount,
          comboLabel,
        });
      }
    }
  }

  return candidates
    .sort((a, b) => b.score - a.score)
    .slice(0, maxSuggestions);
}

/**
 * Génère les meilleures suggestions pour les 3 blocs de la main.
 * Chaque suggestion inclut un potentiel combo/setup pour l'affichage.
 */
export function generateSuggestions(
  grid: Grid,
  hand: Array<BlockInstance | null>,
  maxPerBlock = 3
): Record<number, Suggestion[]> {
  const boolGrid = gridToBool(grid);
  const result: Record<number, Suggestion[]> = {};

  hand.forEach((block, slotIndex) => {
    if (!block) { result[slotIndex] = []; return; }
    result[slotIndex] = suggestionsForBlock(block, boolGrid, maxPerBlock);
  });

  return result;
}

/**
 * Vérifie si un bloc peut être placé quelque part sur la grille.
 */
export function canBlockBePlaced(grid: Grid, block: BlockInstance): boolean {
  const boolGrid = gridToBool(grid);
  const transforms = getUniqueTransforms(block.definition, block.color);
  for (const transform of transforms) {
    const shape = getInstanceShape(transform);
    for (let row = 0; row < GRID_SIZE; row++) {
      for (let col = 0; col < GRID_SIZE; col++) {
        if (canPlace(boolGrid, shape, row, col)) return true;
      }
    }
  }
  return false;
}

/**
 * Suggère automatiquement 3 nouveaux blocs pour recharger la main.
 * Critères : compatibles avec la grille, diversifiés, maximisant le potentiel de combos futurs.
 */
export function suggestNextBlocks(
  grid: Grid,
  colors: string[],
  count = 3
): BlockInstance[] {
  const boolGrid = gridToBool(grid);

  // Scorer chaque bloc du catalogue sur son MEILLEUR placement possible
  const scored = BLOCK_CATALOG.map((def, idx) => {
    const color = colors[idx % colors.length] || colors[0];
    const transforms = getUniqueTransforms(def, color);

    let bestTotalScore = -Infinity;
    let bestSetup = 0;

    for (const instance of transforms) {
      const shape = getInstanceShape(instance);
      for (let row = 0; row < GRID_SIZE; row++) {
        for (let col = 0; col < GRID_SIZE; col++) {
          if (!canPlace(boolGrid, shape, row, col)) continue;
          const s = scoreCandidate(boolGrid, shape, row, col, color);
          if (s.totalScore > bestTotalScore) {
            bestTotalScore = s.totalScore;
            bestSetup = s.nearCompleteCount;
          }
        }
      }
    }

    if (bestTotalScore === -Infinity) return { def, color, totalScore: -1000, setup: 0 };
    return { def, color, totalScore: bestTotalScore + bestSetup * 5, setup: bestSetup };
  });

  scored.sort((a, b) => b.totalScore - a.totalScore);

  // Sélectionner `count` blocs diversifiés (taille et série variées)
  const selected: BlockInstance[] = [];
  const usedSeries = new Set<string>();
  const sizes: number[] = [];

  for (const { def, color } of scored) {
    if (selected.length >= count) break;
    // Favoriser la diversité des séries
    const seriesSaturated = usedSeries.has(def.series) && selected.length < count - 1;
    // Favoriser la diversité des tailles (petit/moyen/grand)
    const sizeSaturated = sizes.filter(s => s === def.size).length >= 2;
    if (seriesSaturated || sizeSaturated) continue;
    usedSeries.add(def.series);
    sizes.push(def.size);
    selected.push({ definition: def, rotation: 0, flipped: false, color });
  }

  // Compléter si pas assez
  if (selected.length < count) {
    for (const { def, color } of scored) {
      if (selected.length >= count) break;
      if (!selected.some(s => s.definition.id === def.id)) {
        selected.push({ definition: def, rotation: 0, flipped: false, color });
      }
    }
  }

  return selected;
}

export function findBossSlot(hand: Array<BlockInstance | null>): number {
  let maxSize = -1;
  let bossIdx = 1; // par défaut : milieu
  hand.forEach((block, idx) => {
    if (block && block.definition.size > maxSize) {
      maxSize = block.definition.size;
      bossIdx = idx;
    }
  });

}

/**

 * Cherche la position qui maximise le potentiel de combo futur pour le boss.
 * Retourne null si aucune position valide n'existe.
 */
export function computeBossReservation(
  grid: Grid,
  bossBlock: BlockInstance,
  bossSlot: number,
  memory: ClearMemory[] = []
): BossReservation | null {
  const boolGrid = gridToBool(grid);
  const transforms = getUniqueTransforms(bossBlock.definition, bossBlock.color);

  let bestScore = -Infinity;
  let bestResult: BossReservation | null = null;

    for (let row = 0; row < GRID_SIZE; row++) {
      for (let col = 0; col < GRID_SIZE; col++) {
        if (!canPlace(boolGrid, shape, row, col)) continue;

        const s = scoreCandidate(boolGrid, shape, row, col, bossBlock.color, new Set(), memory, true);

        }

    }
  }

}

/**
 * Génère les meilleures suggestions pour les 3 blocs de la main.
 * Respecte la zone réservée du boss et intègre la mémoire contextuelle.
 */
export function generateSuggestions(
  grid: Grid,
  hand: Array<BlockInstance | null>,
  maxPerBlock = 3,
  bossReservation: BossReservation | null = null,
  memory: ClearMemory[] = []
): Record<number, Suggestion[]> {
  const boolGrid = gridToBool(grid);
  const result: Record<number, Suggestion[]> = {};
  const reservedCells = reservationToCellSet(bossReservation);
  const bossSlot = bossReservation?.bossSlot ?? findBossSlot(hand);

  hand.forEach((block, slotIndex) => {
    if (!block) { result[slotIndex] = []; return; }
    const isBoss = slotIndex === bossSlot;
    result[slotIndex] = suggestionsForBlock(block, boolGrid, maxPerBlock, reservedCells, memory, isBoss);
  });

  return result;
}

/**
 * Vérifie si un bloc peut être placé quelque part sur la grille.
 */
export function canBlockBePlaced(grid: Grid, block: BlockInstance): boolean {
  const boolGrid = gridToBool(grid);
  const transforms = getUniqueTransforms(block.definition, block.color);
  for (const transform of transforms) {
    const shape = getInstanceShape(transform);
    for (let row = 0; row < GRID_SIZE; row++) {
      for (let col = 0; col < GRID_SIZE; col++) {
        if (canPlace(boolGrid, shape, row, col)) return true;
      }
    }
  }
  return false;
}

/**
 * Suggère automatiquement 3 nouveaux blocs pour recharger la main.
 * GARANTIT que le slot 1 (milieu) reçoit le plus grand bloc (le nouveau boss).
 * Slots 0 et 2 reçoivent des blocs plus petits compatibles.
 */
export function suggestNextBlocks(
  grid: Grid,
  colors: string[],
  count = 3
): BlockInstance[] {
  const boolGrid = gridToBool(grid);

  // Scorer chaque bloc du catalogue sur son MEILLEUR placement possible
  const scored = BLOCK_CATALOG.map((def, idx) => {
    const color = colors[idx % colors.length] || colors[0];
    const transforms = getUniqueTransforms(def, color);

    let bestTotalScore = -Infinity;
    let bestSetup = 0;

    for (const instance of transforms) {
      const shape = getInstanceShape(instance);
      for (let row = 0; row < GRID_SIZE; row++) {
        for (let col = 0; col < GRID_SIZE; col++) {
          if (!canPlace(boolGrid, shape, row, col)) continue;
          const s = scoreCandidate(boolGrid, shape, row, col, color);
          if (s.totalScore > bestTotalScore) {
            bestTotalScore = s.totalScore;
            bestSetup = s.nearCompleteCount;
          }
        }
      }
    }

    if (bestTotalScore === -Infinity) return { def, color, totalScore: -1000, setup: 0 };
    return { def, color, totalScore: bestTotalScore + bestSetup * 5, setup: bestSetup };
  });

  scored.sort((a, b) => b.totalScore - a.totalScore);

  // Sélectionner `count` blocs diversifiés (taille et série variées)
  const selected: BlockInstance[] = [];
  const usedSeries = new Set<string>();
  const sizes: number[] = [];

  for (const { def, color } of scored) {
    if (selected.length >= count) break;
    const seriesSaturated = usedSeries.has(def.series) && selected.length < count - 1;
    const sizeSaturated = sizes.filter(s => s === def.size).length >= 2;
    if (seriesSaturated || sizeSaturated) continue;
    usedSeries.add(def.series);
    sizes.push(def.size);
    selected.push({ definition: def, rotation: 0, flipped: false, color });
  }

  // Compléter si pas assez
  if (selected.length < count) {
    for (const { def, color } of scored) {
      if (selected.length >= count) break;
      if (!selected.some(s => s.definition.id === def.id)) {
        selected.push({ definition: def, rotation: 0, flipped: false, color });
      }
    }
  }

  // ── RÈGLE DU BOSS ──
  // Garantir que le slot 1 (milieu) est toujours le plus GRAND bloc.
  // Trier selected par taille : [petit, grand, moyen] → mettre le plus grand au centre.
  if (selected.length === 3) {
    selected.sort((a, b) => a.definition.size - b.definition.size);
    // Après tri ascendant : [0]=petit, [1]=moyen, [2]=grand
    // On veut : [0]=petit, [1]=grand(boss), [2]=moyen
    const [small, medium, large] = selected;
    return [small, large, medium];
  }

  return selected;
}

/**
 * Suggère un seul nouveau bloc pour un slot non-boss (shuffle).
 * Compatible avec l'état de la grille et ne bloque pas la zone réservée du boss.
 */
export function suggestOneBlock(
  grid: Grid,
  colors: string[],
  excludeIds: string[] = [],
  bossReservation: BossReservation | null = null,
  memory: ClearMemory[] = []
): BlockInstance | null {
  const boolGrid = gridToBool(grid);
  const reservedCells = reservationToCellSet(bossReservation);

  const scored = BLOCK_CATALOG
    .filter(def => !excludeIds.includes(def.id))
    .map((def, idx) => {
      const color = colors[idx % colors.length] || colors[0];
      const transforms = getUniqueTransforms(def, color);

      let bestScore = -Infinity;
      for (const instance of transforms) {
        const shape = getInstanceShape(instance);
        for (let row = 0; row < GRID_SIZE; row++) {
          for (let col = 0; col < GRID_SIZE; col++) {
            if (!canPlace(boolGrid, shape, row, col)) continue;
            const s = scoreCandidate(boolGrid, shape, row, col, color, reservedCells, memory, false);
            if (s.totalScore > bestScore) bestScore = s.totalScore;
          }
        }
      }

      if (bestScore === -Infinity) return null;
      return { def, color, score: bestScore };
    })
    .filter((x): x is { def: typeof BLOCK_CATALOG[0]; color: string; score: number } => x !== null);

  scored.sort((a, b) => b.score - a.score);
  if (scored.length === 0) return null;

  const pick = scored[0];
  return { definition: pick.def, rotation: 0, flipped: false, color: pick.color };
}

  bossSlot = 1,

    const rotations: Array<0 | 90 | 180 | 270> = [0, 90, 180, 270];
    const currentIdx = rotations.indexOf(block.rotation);
    const nextRotation = rotations[(currentIdx + 1) % 4];
    onSlotChange(slot, { ...block, rotation: nextRotation });
  };

  const handleFlip = (slot: 0 | 1 | 2) => {
    const block = hand[slot];
    if (!block) return;
    onSlotChange(slot, { ...block, flipped: !block.flipped });
  };

  return (

      <div className="grid grid-cols-3 gap-2">
        {([0, 1, 2] as const).map(slot => (
          <HandSlot
            key={slot}
            slot={slot}
            block={hand[slot]}
            onSelect={(s, b) => onSlotChange(s, b)}
            onRemove={s => onSlotChange(s, null)}
            onRotate={handleRotate}
            onFlip={handleFlip}
            isBoss={slot === bossSlot}
            onShuffle={slot !== bossSlot ? onShuffleSlot : undefined}
            isHighlighted={highlightedSlot === slot}
            isKeyboardSelected={keyboardSelectedSlot === slot}
          />
        ))}
      </div>
      {keyboardSelectedSlot !== null && keyboardSelectedSlot !== undefined && (
        <div className="text-xs text-primary text-center animate-suggestion-in">
          ⌨️ Bloc {keyboardSelectedSlot + 1} sélectionné — Utilisez les flèches sur la grille, Entrée pour placer, Échap pour annuler
        </div>
      )}
      {(keyboardSelectedSlot === null || keyboardSelectedSlot === undefined) && (
        <div className="text-xs text-muted-foreground text-center">
          Glissez un bloc vers la grille ou appuyez sur{' '}
          <kbd className="px-1 py-0.5 rounded bg-muted font-mono text-[10px]">1</kbd>
          <kbd className="px-1 py-0.5 rounded bg-muted font-mono text-[10px] mx-0.5">2</kbd>
          <kbd className="px-1 py-0.5 rounded bg-muted font-mono text-[10px]">3</kbd> pour sélectionner un slot
